defmodule Cmp do
  @moduledoc "Fixture comparison module."

  def ordered?(a, b) do
    case :persistent_term.get(Mut.Runtime.active_key(), 0) do
      0 -> a <= b
      11 -> a > b
      20 -> a < b
      _ -> a <= b
    end
  end

  def distinct_direction?(a, b) do
    mutalisk_schema_active = :persistent_term.get(Mut.Runtime.active_key(), 0)

    case mutalisk_schema_active do
      0 ->
        case mutalisk_schema_active do
          0 ->
            case mutalisk_schema_active do
              0 ->
                case mutalisk_schema_active do
                  0 -> a > b
                  24 -> a >= b
                  29 -> a <= b
                  _ -> a > b
                end or
                  case mutalisk_schema_active do
                    0 -> a >= b
                    6 -> a > b
                    9 -> a < b
                    _ -> a >= b
                  end

              12 ->
                a > b and a >= b

              _ ->
                case mutalisk_schema_active do
                  0 -> a > b
                  24 -> a >= b
                  29 -> a <= b
                  _ -> a > b
                end or
                  case mutalisk_schema_active do
                    0 -> a >= b
                    6 -> a > b
                    9 -> a < b
                    _ -> a >= b
                  end
            end or
              case mutalisk_schema_active do
                0 -> a == b
                30 -> a != b
                _ -> a == b
              end

          16 ->
            (a > b or a >= b) and a == b

          _ ->
            case mutalisk_schema_active do
              0 ->
                case mutalisk_schema_active do
                  0 -> a > b
                  24 -> a >= b
                  29 -> a <= b
                  _ -> a > b
                end or
                  case mutalisk_schema_active do
                    0 -> a >= b
                    6 -> a > b
                    9 -> a < b
                    _ -> a >= b
                  end

              12 ->
                a > b and a >= b

              _ ->
                case mutalisk_schema_active do
                  0 -> a > b
                  24 -> a >= b
                  29 -> a <= b
                  _ -> a > b
                end or
                  case mutalisk_schema_active do
                    0 -> a >= b
                    6 -> a > b
                    9 -> a < b
                    _ -> a >= b
                  end
            end or
              case mutalisk_schema_active do
                0 -> a == b
                30 -> a != b
                _ -> a == b
              end
        end or
          case mutalisk_schema_active do
            0 -> a != b
            1 -> a == b
            _ -> a != b
          end

      25 ->
        (a > b or a >= b or a == b) and a != b

      _ ->
        case mutalisk_schema_active do
          0 ->
            case mutalisk_schema_active do
              0 ->
                case mutalisk_schema_active do
                  0 -> a > b
                  24 -> a >= b
                  29 -> a <= b
                  _ -> a > b
                end or
                  case mutalisk_schema_active do
                    0 -> a >= b
                    6 -> a > b
                    9 -> a < b
                    _ -> a >= b
                  end

              12 ->
                a > b and a >= b

              _ ->
                case mutalisk_schema_active do
                  0 -> a > b
                  24 -> a >= b
                  29 -> a <= b
                  _ -> a > b
                end or
                  case mutalisk_schema_active do
                    0 -> a >= b
                    6 -> a > b
                    9 -> a < b
                    _ -> a >= b
                  end
            end or
              case mutalisk_schema_active do
                0 -> a == b
                30 -> a != b
                _ -> a == b
              end

          16 ->
            (a > b or a >= b) and a == b

          _ ->
            case mutalisk_schema_active do
              0 ->
                case mutalisk_schema_active do
                  0 -> a > b
                  24 -> a >= b
                  29 -> a <= b
                  _ -> a > b
                end or
                  case mutalisk_schema_active do
                    0 -> a >= b
                    6 -> a > b
                    9 -> a < b
                    _ -> a >= b
                  end

              12 ->
                a > b and a >= b

              _ ->
                case mutalisk_schema_active do
                  0 -> a > b
                  24 -> a >= b
                  29 -> a <= b
                  _ -> a > b
                end or
                  case mutalisk_schema_active do
                    0 -> a >= b
                    6 -> a > b
                    9 -> a < b
                    _ -> a >= b
                  end
            end or
              case mutalisk_schema_active do
                0 -> a == b
                30 -> a != b
                _ -> a == b
              end
        end or
          case mutalisk_schema_active do
            0 -> a != b
            1 -> a == b
            _ -> a != b
          end
    end
  end
