defmodule Bool do
  @moduledoc "Fixture boolean module."

  def strict?(a, b) do
    mutalisk_schema_active = :persistent_term.get(Mut.Runtime.active_key(), 0)

    case mutalisk_schema_active do
      0 ->
        a and
          case mutalisk_schema_active do
            0 -> not b
            2 -> b
            _ -> not b
          end

      22 ->
        a or not b

      _ ->
        a and
          case mutalisk_schema_active do
            0 -> not b
            2 -> b
            _ -> not b
          end
    end
  end

  def loose?(a, b) do
    mutalisk_schema_active = :persistent_term.get(Mut.Runtime.active_key(), 0)

    case mutalisk_schema_active do
      0 ->
        case mutalisk_schema_active do
          0 -> a || b
          23 -> a && b
          _ -> a || b
        end ||
          case mutalisk_schema_active do
            0 -> !b
            8 -> b
            _ -> !b
          end

      17 ->
        (a || b) && !b

      _ ->
        case mutalisk_schema_active do
          0 -> a || b
          23 -> a && b
          _ -> a || b
        end ||
          case mutalisk_schema_active do
            0 -> !b
            8 -> b
            _ -> !b
          end
    end
  end
end
