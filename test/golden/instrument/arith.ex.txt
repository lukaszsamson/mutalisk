defmodule Arith do
  @moduledoc "Fixture arithmetic module."

  def score(a, b) do
    mut_active = :persistent_term.get(Mut.Runtime.active_key(), 0)

    case mut_active do
      0 ->
        case mut_active do
          0 -> a + b
          4 -> a * b
          28 -> a - b
          _ -> a + b
        end *
          case mut_active do
            0 -> a - b
            6 -> a * b
            8 -> a + b
            _ -> a - b
          end

      3 ->
        (a + b) / (a - b)

      18 ->
        a + b + (a - b)

      19 ->
        a + b - (a - b)

      _ ->
        case mut_active do
          0 -> a + b
          4 -> a * b
          28 -> a - b
          _ -> a + b
        end *
          case mut_active do
            0 -> a - b
            6 -> a * b
            8 -> a + b
            _ -> a - b
          end
    end
  end

  def integer_parts(a, b) do
    mut_active = :persistent_term.get(Mut.Runtime.active_key(), 0)

    {case mut_active do
       0 -> div(a, b)
       27 -> rem(a, b)
       _ -> div(a, b)
     end,
     case mut_active do
       0 -> rem(a, b)
       17 -> div(a, b)
       _ -> rem(a, b)
     end}
  end
end
