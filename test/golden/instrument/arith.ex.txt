defmodule Arith do
  @moduledoc "Fixture arithmetic module."

  def score(a, b) do
    mutalisk_schema_active = :persistent_term.get(Mut.Runtime.active_key(), 0)

    case mutalisk_schema_active do
      0 ->
        case mutalisk_schema_active do
          0 -> a + b
          4 -> a * b
          28 -> a - b
          _ -> a + b
        end *
          case mutalisk_schema_active do
            0 -> a - b
            5 -> a * b
            7 -> a + b
            _ -> a - b
          end

      3 ->
        (a + b) / (a - b)

      19 ->
        a + b + (a - b)

      21 ->
        a + b - (a - b)

      _ ->
        case mutalisk_schema_active do
          0 -> a + b
          4 -> a * b
          28 -> a - b
          _ -> a + b
        end *
          case mutalisk_schema_active do
            0 -> a - b
            5 -> a * b
            7 -> a + b
            _ -> a - b
          end
    end
  end

  def integer_parts(a, b) do
    mutalisk_schema_active = :persistent_term.get(Mut.Runtime.active_key(), 0)

    {case mutalisk_schema_active do
       0 -> div(a, b)
       27 -> rem(a, b)
       _ -> div(a, b)
     end,
     case mutalisk_schema_active do
       0 -> rem(a, b)
       18 -> div(a, b)
       _ -> rem(a, b)
     end}
  end
end
